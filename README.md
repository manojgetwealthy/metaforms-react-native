## Install

```bash
npm install --save metaforms-react-native
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'metaforms-react-native'
import 'metaforms-react-native/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
